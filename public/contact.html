# Generate contact information for Saif Rahman
# http://www.clomen.web.app
 import os
import sys
import numpy
import string

if sys.version > '3':
strl = str
else:
strl = string

EMAIL = """
\u0073\u0061\u0069\u0066\u0072\u0061\u0068\u006d\u006e\u0037\u0038\u0040\u0067\u006d\u0061\u0069\u006c\u002e\u0063\u006f\u006d
"""

y = string.ascii_letters
n = len(y)
M = numpy.identity(n, dtype=numpy.int32)
M[:n, n-1] = 1

a = numpy.array(list(map(ord, y)), dtype=numpy.int32)
x = ''.join(map(chr, numpy.dot(M, a)))
uncipher = strl.maketrans(x,y)

print( strl.translate(EMAIL, uncipher) )
